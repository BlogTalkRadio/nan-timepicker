NaN Time-picker

A lightweight time-picker plugin based on 'jQuery timepicker plugin'.

Basic Example

$('#basicExample').timepicker();

Scroll Default Example

Set the scroll position to local time if no value selected.

$('#defaultValueExample').timepicker({ 'scrollDefaultNow': true });

DisableTimeRanges Example

Prevent selection of certain time values.

$('#disableTimeRangesExample').timepicker( { 'className': 'no-scroll', 'disableTimeRanges': [[0, 10800], ['7pm', '11pm']] });

DisableTimeRanges Example

Prevent selection of certain time values.

$('#disableTimeRangesExample2').timepicker( { 'disableTimeRanges': [['1am', '2am'], ['3am', '4:01am']] });

DisableTimeRanges Example

Prevent selection of certain time values.

$('#disableTimeRangesExample3').timepicker( { className: 'no-scroll', disableTimeRanges: [[0, 10800], ['7pm', '11pm']], timeRanges: [ { className: 'highlighted', disabled: true, ranges: [['5am', '6am'], ['11am', '12am']] }, { className: 'highlighted2', disabled: false, ranges: [['2pm', '3pm'], ['6pm', '7pm']] } ] });
